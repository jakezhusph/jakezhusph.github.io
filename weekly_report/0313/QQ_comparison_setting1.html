KDC, save results, everyother

rm(list=ls())
source("/net/mulan/disk2/jiaqiang/kdc/code/v5/kdc.R")

outpath = "/net/mulan/disk2/jiaqiang/kdc/experiment/simulation/v4/output/kdc/"
respath = "/net/mulan/disk2/jiaqiang/kdc/experiment/simulation/v4/result/"

numZero = 2000
sigZero = 200
layerlist <- c("E","ONL","GL")
for(iblock in c(25,20,15,10)){
    load(paste0("/net/mulan/disk2/jiaqiang/kdc/experiment/realdata/hdst/data/CN24_E1_",iblock,"X_new_barcode_with_layers.rds"))
    cat(paste0(iblock,"X: number of Spots ",nrow(new_barcode_layer),"\n"))
    kdc_pval <- list()
    for(ilayer in 1:3){
        single_pval <- list()
        for(isid in 1:10){
            allpval <- c()
            # cosine
            for(iker in 1:5){
                load(paste0(outpath,"/sim_kdc_CN24_E1_",iblock,"X_",layerlist[ilayer],"_everyother_cos",iker,"_zero",numZero,"_sigzero",sigZero,"_rpt",isid,".rds"))

                allpval <- cbind(allpval,res$pval)
                rm(res)
            }
            # gaussian
            for(iker in 1:5){
                load(paste0(outpath,"/sim_kdc_CN24_E1_",iblock,"X_",layerlist[ilayer],"_everyother_gaus",iker,"_zero",numZero,"_sigzero",sigZero,"_rpt",isid,".rds"))
                allpval <- cbind(allpval,res$pval)
                rm(res)
            }

            # direct projection
            load(paste0(outpath,"/sim_kdc_CN24_E1_",iblock,"X_",layerlist[ilayer],"_projection_zero",numZero,"_sigzero",sigZero,"_rpt",isid,".rds")) 
            allpval <- cbind(allpval,res$pval)
            colnames(allpval) <- c(paste0("cos",1:5),paste0("gaus",1:5),"projection")
            rownames(allpval) <- paste0("gene",1:1000)
            single_pval[[isid]] <- allpval
            rm(res,allpval)
        }

        kdc_pval[[ilayer]] <- single_pval
        rm(single_pval)
    }
    names(kdc_pval) <- layerlist
    save(kdc_pval,file=paste0(respath,"/sim_kdc_CN24_E1_",iblock,"X_zero",numZero,"_sigzero",sigZero,"_everyother_pvals.rds"))
}

rm(list=ls())
outpath = "/net/mulan/disk2/jiaqiang/kdc/experiment/simulation/v4/output/spatialde/"
respath = "/net/mulan/disk2/jiaqiang/kdc/experiment/simulation/v4/result/"

numZero = 2000
sigZero = 200

layerlist <- c("E","ONL","GL")
for(iblock in c(25,20,15)){
    load(paste0("/net/mulan/disk2/jiaqiang/kdc/experiment/realdata/hdst/data/CN24_E1_",iblock,"X_new_barcode_with_layers.rds"))
    cat(paste0(iblock,"X: number of Spots ",nrow(new_barcode_layer),"\n"))
    spe_pval <- list()
    for(ilayer in 1:3){
        single_pval <- c()
        for(isid in 1:10){
            res     <- read.table(paste0(outpath,"/sim_spe_CN24_E1_",iblock,"X_",layerlist[ilayer],"_zero",numZero,"_sigzero",sigZero,"_rpt",isid,".csv"),head=T)
            res1    <- res[-which(res$g=="log_total_count"),]

            if(sum(duplicated(res1$g))>0){
                clean_res <- res1[-which(duplicated(res1$g)),]
            }else{
                clean_res <- res1
            }

            order_res <- clean_res[order(as.numeric(sapply(strsplit(as.character(clean_res$g),split="gene"),"[[",2))),]
            single_pval <- cbind(single_pval,order_res$pval)
            rm(res,res1,order_res,clean_res)
        }
        colnames(single_pval) <- paste0("rpt",1:10)
        rownames(single_pval) <- paste0("gene",1:1000)
        spe_pval[[ilayer]] <- single_pval
    }
    names(spe_pval) <- layerlist
    save(spe_pval,file=paste0(respath,"/sim_spe_CN24_E1_",iblock,"X_zero",numZero,"_sigzero",sigZero,"_pvals.rds"))
}
