rm(list=ls())
source("/net/mulan/disk2/jiaqiang/kdc/code/v5/kdc.R")
source("/net/fantasia/home/jiaqiang/mulan_temp/common_function/simple_gg_general.R")
outpath = "/net/mulan/disk2/jiaqiang/kdc/experiment/realdata/hdst/output/"
respath = "/net/mulan/disk2/jiaqiang/kdc/experiment/simulation/v3/result/"

allpval <- list()
allpval_proj <- list()
for(isid in 1:10){
    single_per_pval <- c()
    for(iker in 1:10){
        load(paste0(outpath,"/CN24_E1_unmodgtf_filtered_red_ut_HDST_5x_cos",iker,"_perm",isid,".rds"))
        single_per_pval <- cbind(single_per_pval,res$pval)
        rm(res)
    }

    for(iker in 1:10){
        load(paste0(outpath,"/CN24_E1_unmodgtf_filtered_red_ut_HDST_5x_gaus",iker,"_perm",isid,".rds"))
        single_per_pval <- cbind(single_per_pval,res$pval)
        rm(res)
    }

    load(paste0(outpath,"/CN24_E1_unmodgtf_filtered_red_ut_HDST_5x_Projection_perm",isid,".rds"))
    single_per_pval <- cbind(single_per_pval,res$pval)
    rm(res)
    combined_pvals  <- CombinePValues(single_per_pval)
    allpval[[isid]] <- combined_pvals
    allpval_proj[[isid]] <- single_per_pval[,21]
    rm(combined_pvals,single_per_pval)
}

library(ggpubr)
fig1 <- qplot_gg(pvalues=unlist(allpval),
                legend.position="none",pt.size=2,ax.txt.size=10,
                ax.title.size=10,len.txt.size=1,cl=1) 

fig2 <- qplot_gg(pvalues=unlist(allpval_proj),
                legend.position="none",pt.size=2,ax.txt.size=10,
                ax.title.size=10,len.txt.size=1,cl=1) 

two_figs <- ggarrange(fig1+ggtitle("combined"),fig2+ggtitle("projection"),
                ncol = 2, nrow = 1)
