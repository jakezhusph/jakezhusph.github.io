cos_trans <- function(x,lkernel=5){
    l   <- quantile(x,probs=seq(0.1,1,by=0.1))
    out <- cos(2*pi*x/l[lkernel])
    return(out)
}

iblock = 25
load(paste0(simpath,"/sim_data_CN24_E1_",iblock,"X_ONL_zero5_sigzero5_rpt5.rds"))

newx <- seq(min(location[,1]),max(location[,1]),length.out=1000)
newy <- seq(min(location[,2]),max(location[,2]),length.out=1000)


raw_loc     <- cbind(x=newx,y=newy)
cor_list <- cov_list <- list()
for(iker in 1:10){
    trans_loc   <- apply(raw_loc,2,cos_trans,lkernel=iker)
    scale_trans_loc     <- apply(trans_loc,2,scale)
    order_loc       <- scale_trans_loc[order(scale_trans_loc[,1],scale_trans_loc[,2]),]
    kmat        <- order_loc %*% solve(t(order_loc) %*% order_loc) %*% t(order_loc)
    a           <- cov2cor(kmat)
    cor_list[[iker]] <- ht_map_K_uniform(a,lgd=F,main=paste0("Cosine_Correlation",iker))
    cov_list[[iker]] <- ht_map_K_uniform(kmat,lgd=F,main=paste0("Cosine_Covariance",iker))
    rm(a,kmat,trans_loc,scale_trans_loc,order_loc)
}



library(ggpubr)
fig3 <- ggarrange(cor_list[[1]],cor_list[[2]],

                cor_list[[9]],cor_list[[10]],

gaus_trans <- function(x,lkernel=5){
    l   <- quantile(x,probs=seq(0.1,1,by=0.1))
    out <- exp(-x^2/(2*l[lkernel]^2))
    return(out)
}


iblock = 25

for(iker in 1:10){
