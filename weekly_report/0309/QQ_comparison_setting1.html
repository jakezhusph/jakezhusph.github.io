    load(paste0("/net/mulan/disk2/jiaqiang/kdc/experiment/realdata/hdst/data/CN24_E1_",iblock,"X_new_barcode_with_layers.rds"))
    cat(paste0(iblock,"X: number of Spots ",nrow(new_barcode_layer),"\n"))
    spe_pval <- list()
    for(ilayer in 1:4){
        single_pval <- c()
        for(isid in 1:10){
            res     <- read.table(paste0(outpath,"/sim_spe_CN24_E1_",iblock,"X_",layerlist[ilayer],"_zero",numZero,"_sigzero",sigZero,"_rpt",isid,".csv"),head=T)
            res1    <- res[-which(res$g=="log_total_count"),]

            if(sum(duplicated(res1$g))>0){
                clean_res <- res1[-which(duplicated(res1$g)),]
            }else{
                clean_res <- res1
            }

            order_res <- clean_res[order(as.numeric(sapply(strsplit(as.character(clean_res$g),split="gene"),"[[",2))),]
            single_pval <- cbind(single_pval,order_res$pval)
            rm(res,res1,order_res,clean_res)
        }
        colnames(single_pval) <- paste0("rpt",1:10)
        rownames(single_pval) <- paste0("gene",1:1000)
        spe_pval[[ilayer]] <- single_pval
    }
    names(spe_pval) <- layerlist
    save(spe_pval,file=paste0(respath,"/sim_spe_CN24_E1_",iblock,"X_zero",numZero,"_sigzero",sigZero,"_pvals.rds"))
}

    load(paste0(respath,"/sim_spark_CN24_E1_",iblock,"X_zero10_sigzero1_pvals.rds"))

    for(ilayer in 1:4){

        if(ilayer==1){
            single_sample_qq_fig[[ilayer]] <- qplot_gg(pvalues=list(KDC=as.vector(single_kdc_pval),

                                        legend.position=c(0.25,0.9),pt.size=2,ax.txt.size=10,ax.title.size=10,len.txt.size=1,cl=1,
                                        factor_level=c("KDC","SpatialDE","SPARK"),self_label=c("KDC","SpatialDE","SPARK")) 
        }else{
            single_sample_qq_fig[[ilayer]] <- qplot_gg(pvalues=list(KDC=as.vector(single_kdc_pval),
                                    SpatialDE=as.vector(spe_pval[[ilayer]][101:1000,]),
