# Cosine 
cos_trans <- function(x,lkernel=5){
    l   <- quantile(x,probs=seq(0.1,0.9,by=0.2))
    out <- cos(2*pi*x/l[lkernel])
    return(out)
}

for(isam in 1:6){
    load(paste0(workdir,"/data/",sample[isam],"_unmodgtf_filtered_red_ut_",itype,"_final.rds"))
    raw_loc     <- as.matrix(location[,1:2])
    for(iker in 1:5){
        trans_loc   <- apply(raw_loc,2,cos_trans,lkernel=iker)
        res         <- kdc_spark(sp_count,trans_loc,mc_cores=20)
        save(res,file=paste0(workdir,"/output/",sample[isam],"_",itype,"_everyother_cosine",iker,"_kdc.rds"))
        rm(res,trans_loc)
    }
    rm(raw_loc,sp_count)
}


# Gaus 
gaus_trans <- function(x,lkernel=5){
    l   <- quantile(x,probs=seq(0.1,0.9,by=0.2))
    out <- exp(-x^2/(2*l[lkernel]^2))

    return(out)
}

for(isam in 1:6){
    load(paste0(workdir,"/data/",sample[isam],"_unmodgtf_filtered_red_ut_",itype,"_final.rds"))
    raw_loc     <- as.matrix(location[,1:2])
    for(iker in 1:5){
        trans_loc   <- apply(raw_loc,2,gaus_trans,lkernel=iker)
        res         <- kdc_spark(sp_count,trans_loc,mc_cores=20)
        save(res,file=paste0(workdir,"/output/",sample[isam],"_",itype,"_everyother_gaus",iker,"_kdc.rds"))
        rm(res,trans_loc)
    }
    rm(raw_loc,sp_count)
}
